//Program to convert a string into toggle case
#include &lt;stdio.h&gt;
int main()
{
char str[100];
scanf(&quot;%[^\n]s&quot;, str);
for (int i = 0; str[i];i++){
if (str[i]&gt;=&#39;A&#39; &amp;&amp; str[i]&lt;=&#39;Z&#39;)
str[i] = str[i] + 32; //converting upper case into lower case
else if (str[i]&gt;=&#39;a&#39; &amp;&amp; str[i]&lt;=&#39;z&#39;)
str[i] -= 32; //converting lower case into upper case
}
puts(str);
return 0;
}
